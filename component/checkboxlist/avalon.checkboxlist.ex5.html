异步改变checkboxlist data

onSelect获取values: {{el}}  

<!DOCTYPE html>
<html>
    
    <head>
        <title>checkboxlist demo 5</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"/>
        <script src="../../avalon.js"></script>
        <script src="../style/highlight/shCore.js"></script>
        <script>
            avalon.library("oni", {
                $ready: function () {
                    avalon.log("控件已经构建完毕")
                }
            })
            require(["./component/checkboxlist/avalon.checkboxlist"], function(avalon) {
                var demoVM = avalon.define({
                    $id: "demo",
                    checkedInputs: [],
                    opts: {
                        data: [
                            { text : '文字1' , value : 'w1' } ,
                            { text : '文字2' , value : 'w2' } ,
                            { text : '文字3' , value : 'w3' } ,
                            { text : '文字4' , value : 'w4' }
                        ],
                        onSelect: function(e, inputsData){
                            demoVM.checkedInputs = inputsData
                        }
                    }
                })
    
                setTimeout(function(){
                    avalon.vmodels.oni_checkboxlist.data = [
                        { text : '文字5' , value : 'w5' } ,
                        { text : '文字6' , value : 'w6' }
                    ]
                }, 2000)
    
                avalon.scan()
            });
        </script>
    </head>
    
    <body>
        <div class="wrapper">
            <div class="demo-show" ms-controller="demo">
                <fieldset>
                    <legend>异步改变checkboxlist data</legend>
                    <oni:checkboxlist $id="oni_checkboxlist"
                    configs="opts" alltext=""></oni:checkboxlist>
                    <p>onSelect获取values:
                        <span ms-repeat="checkedInputs">{{el}} </span>
                    </p>
                </fieldset>
            </div>
            <div ms-skip></div>
        </div>
    </body>

</html>
