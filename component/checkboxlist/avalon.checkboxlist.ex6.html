自定义选项模板 全部选项

onSelect获取values: {{el}}  

<!DOCTYPE html>
<html>
    
    <head>
        <title>checkboxlist demo 6</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"
        />
        <script src="../../avalon.js"></script>
        <script src="../style/highlight/shCore.js"></script>
        <script>
            avalon.library("oni", {
                $ready: function () {
                    avalon.log("控件已经构建完毕")
                }
            })
            require(["./component/checkboxlist/avalon.checkboxlist"], function(avalon) {
                var demoVM = avalon.define({
                    $id: "demo",
                    checkedInputs: [],
                    opts: {
                        type: "week",
                        data: [
                            { text : '选项{{$index + 1}}：<i>文字1</i>' , value : 'w1' } ,
                            { text : '选项{{$index + 1}}：<i>文字2</i>' , value : 'w2' } ,
                            { text : '选项{{$index + 1}}：<i>文字3</i>' , value : 'w3' } ,
                            { text : '选项{{$index + 1}}：<i>文字4</i>' , value : 'w4' }
                        ],
                        onSelect: function(e, inputsData){
                            demoVM.checkedInputs = inputsData
                        }
                    }
                })
                avalon.scan()
            });
        </script>
    </head>
    
    <body>
        <div class="wrapper">
            <div class="demo-show" ms-controller="demo">
                <fieldset>
                    <legend>自定义选项模板</legend>
                    <oni:checkboxlist configs="opts">
                        <b slot="alltext">全部选项</b>
                    </oni:checkboxlist>
                    <p>onSelect获取values:
                        <span ms-repeat="checkedInputs">{{el}} </span>
                    </p>
                </fieldset>
            </div>
            <div ms-skip></div>
        </div>
    </body>

</html>
