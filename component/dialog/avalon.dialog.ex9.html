修改dialog的title、content

说明：想修改现有弹出层的title, content,直接在$ready方法得到它的title, content修改就行了, 不再提供setTitle, setContent, setModel这样笨重的方法

动态修改dialog的title、content或者重新渲染 可以动态改变对话框的内容{{a}}，点击不同的按钮，试试看它的灵活性呗 
<!DOCTYPE html>
<html>
<head>
<meta http-equiv="content-type" content="text/html;charset=utf-8" />
<meta http-equiv="x-ua-compatible" content="ie=edge" />
<title>dialog demo 8</title>
        <link type="text/css" rel="stylesheet"  href="../highlight/shCore.css"/>
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"/>
        <script src="../highlight/shCore.js"></script>
        <script src="../../avalon.js"></script>
</head>
<body>
        <div class="demo-show" ms-controller="demo">
            <fieldset>  
                <legend>动态修改dialog的title、content或者重新渲染</legend>
                    <button ms-click="show('aa')">打开对话框</button>
                    <button ms-click="changeContent('aa')">改变对话框内容</button>
                    <button ms-click="changeTitle('aa')">改变对话框title</button>
                    <oni:dialog $id="aa" config="$aaOpts">
                        可以动态改变对话框的内容{{a}}，点击不同的按钮，试试看它的灵活性呗
                    </oni:dialog>
            </fieldset>
            <script> 
                require(["./component/dialog/avalon.dialog"], function () {
               
                    avalon.define({
                        $id: "demo",
                        $aaOpts: {
                            title: "title1",
                            showClose: false,
                            $slot: "content",
                            container: document.getElementById("modalWrapper")
                        },
                        a: "这是内容",
                        show: function( id ){
                            avalon.vmodels[id].toggle = true;
                        },
                        changeTitle: function(id){
                            var dialog = avalon.vmodels[id];
                            if(!dialog) {
                                return;
                            }
                            dialog.title = "新标题"
                        },
                        changeContent: function(id){
                            var dialog = avalon.vmodels[id];
                            if(!dialog) {
                                return;
                            }
                           dialog.content = "<strong>{{a}}</strong>"
                        },
                    })
                    avalon.scan();
                })
               
            </script>
        </div>
    </body>
